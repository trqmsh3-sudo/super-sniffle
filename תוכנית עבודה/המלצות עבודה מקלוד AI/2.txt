// מקורות שמשתמש רגיל לא יכול לגשת אליהם בקלות

class UniqueDataAggregator {
  constructor() {
    // APIs שדורשים מנוי/גישה מיוחדת
    this.amazonAPI = new AmazonProductAPI(process.env.AMAZON_API_KEY);
    this.camelCamelAPI = new CamelCamelAPI(process.env.CAMEL_KEY); // מחירים היסטוריים
    this.keepaAPI = new KeepaAPI(process.env.KEEPA_KEY); // מעקב מחירים
    this.fakeSpotAPI = new FakeSpotAPI(process.env.FAKESPOT_KEY); // ביקורות מזויפות
    this.consumerReportsAPI = process.env.CR_API_KEY; // Consumer Reports (בתשלום)
  }

  async buildEnhancedDossier(productName, category) {
    console.log(`🔍 Building ENHANCED dossier with unique data...`);
    
    // 1️⃣ נתונים שאי אפשר לקבל בחיפוש רגיל:
    
    const [
      amazonData,        // ביקורות מאומתות בלבד
      priceHistory,      // גרף מחירים 12 חודשים אחורה
      fakeReviewScore,   // כמה ביקורות מזויפות?
      warrantyData,      // נתוני תביעות אחריות
      returnRate,        // אחוז החזרות
      consumerReports    // דירוגים מקצועיים
    ] = await Promise.all([
      this.getAmazonVerifiedReviews(productName),
      this.getPriceHistory(productName),
      this.analyzeFakeReviews(productName),
      this.getWarrantyClaims(productName, category),
      this.getReturnRate(productName),
      this.getConsumerReportsData(productName)
    ]);
    
    // 2️⃣ ניתוח מתקדם שדורש זמן/מומחיות:
    
    const analysis = {
      // ✅ ערך אמיתי: מחיר הוגן?
      pricing: {
        current: amazonData.currentPrice,
        lowest12Months: priceHistory.min,
        highest12Months: priceHistory.max,
        average: priceHistory.avg,
        trend: priceHistory.trend, // "rising" / "falling"
        recommendation: this.calculatePricingRecommendation(priceHistory),
        // "Wait 2 weeks - price usually drops 15% in January"
      },
      
      // ✅ ערך אמיתי: האם הביקורות אמיתיות?
      reviewTrustworthiness: {
        totalReviews: amazonData.totalReviews,
        verifiedPurchases: amazonData.verifiedCount,
        suspiciousReviews: fakeReviewScore.suspiciousCount,
        trustScore: fakeReviewScore.trustScore, // 0-100
        warning: fakeReviewScore.warning
        // "⚠️ 23% of reviews appear suspicious - be cautious"
      },
      
      // ✅ ערך אמיתי: כמה אנשים מחזירים?
      reliability: {
        returnRate: returnRate.percentage, // 8.5%
        commonReasons: returnRate.topReasons,
        warrantyClaims: warrantyData.claimRate,
        avgLifespan: warrantyData.avgLifespan,
        failureRate: warrantyData.failureRate
        // "Return rate: 8.5% (industry avg: 5%) - mostly due to battery issues"
      },
      
      // ✅ ערך אמיתי: מה המומחים אומרים?
      professionalReviews: {
        consumerReports: consumerReports.rating,
        expertScore: consumerReports.expertScore,
        labTests: consumerReports.testResults,
        // "Consumer Reports rated it 72/100 - durability concerns in lab tests"
      },
      
      // ✅ ערך אמיתי: השוואה לתחרות
      competitive: await this.compareToCompetitors(productName, category),
      // "Similar products: Galaxy S24 ($50 less, 2% higher return rate)"
    };
    
    return {
      success: true,
      uniqueData: analysis,
      sources: this.listUniqueSources(),
      valueProposition: "Data you can't easily find yourself"
    };
  }

  async getAmazonVerifiedReviews(productName) {
    // Amazon Product Advertising API
    // רק verified purchase reviews
    // גישה למטא-דאטה שלא נראית בחיפוש רגיל
    return {
      currentPrice: 799,
      totalReviews: 12453,
      verifiedCount: 9821,
      avgRating: 4.3,
      reviewDistribution: {
        5: 6234,
        4: 2891,
        3: 1456,
        2: 892,
        1: 980
      }
    };
  }

  async getPriceHistory(productName) {
    // Keepa / CamelCamelCamel API
    // גרף מחירים 12 חודשים
    return {
      current: 799,
      min: 699,
      max: 899,
      avg: 765,
      trend: "falling",
      lastDrop: "2 weeks ago",
      nextExpectedDrop: "estimated in 3 weeks (Black Friday pattern)"
    };
  }

  async analyzeFakeReviews(productName) {
    // Fakespot / ReviewMeta API
    // זיהוי ביקורות מזויפות
    return {
      suspiciousCount: 2864,
      trustScore: 68,
      warning: "23% of reviews show suspicious patterns - incentivized reviews detected"
    };
  }

  async getWarrantyClaims(productName, category) {
    // Consumer Reports / SquareTrade API
    // נתוני תביעות אחריות אמיתיות
    return {
      claimRate: 12.3, // 12.3% filed warranty claims
      avgLifespan: 3.2, // years
      failureRate: 0.08,
      commonIssues: [
        "Battery degradation (45% of claims)",
        "Screen defects (23%)",
        "Charging port failure (18%)"
      ]
    };
  }

  async getReturnRate(productName) {
    // Retailer APIs / Industry data
    return {
      percentage: 8.5,
      industryAvg: 5.2,
      topReasons: [
        "Didn't meet expectations (32%)",
        "Battery life issues (28%)",
        "Defective on arrival (18%)"
      ]
    };
  }

  calculatePricingRecommendation(priceHistory) {
    if (priceHistory.current <= priceHistory.min * 1.05) {
      return "🟢 BUY NOW - Near all-time low";
    }
    if (priceHistory.current >= priceHistory.max * 0.95) {
      return "🔴 WAIT - Price is at peak, likely to drop soon";
    }
    if (priceHistory.trend === "falling") {
      return "🟡 CONSIDER WAITING - Price trending down";
    }
    return "🟢 FAIR PRICE - Within normal range";
  }

  listUniqueSources() {
    return [
      "Amazon Product Advertising API (verified reviews only)",
      "Keepa Price History (12-month tracking)",
      "Fakespot Review Analysis",
      "Consumer Reports Lab Tests",
      "Retailer Return Rate Data",
      "Warranty Claim Statistics"
    ];
  }
}

export default UniqueDataAggregator;

/*
❌ חיפוש רגיל ב-Google:
   - ביקורות (כולל מזויפות)
   - מחירים נוכחיים

   - ✅ ביקורות אמיתיות? (ניתוח Fakespot)
   - ✅ אמין? (return rate, warranty claims)
   - ✅ מה המומחים אומרים? (Consumer Reports)
   - ✅ להמתין או לקנות עכשיו? (price prediction)

💰 זה שווה כסף כי:
   1. חוסך זמן (10 מקורות ב-1 מקום)
   2. נתונים שקשה להשיג (APIs בתשלום)
   3. ניתוח מקצועי (fake reviews, pricing patterns)// backend/services/expertAnalyzer.js
// ניתוח שדורש מומחיות ולא רק "חיפוש + סיכום"

class ExpertAnalyzer {
  
  async buildExpertDossier(productName, category, rawData) {
    console.log(`🧠 Running EXPERT analysis for: ${productName}`);
    
    // 1️⃣ ניתוח דפוסים ב-13,000 ביקורות (לא רק קריאה)
    const patternAnalysis = await this.analyzeReviewPatterns(rawData.reviews);
    
    // 2️⃣ זיהוי בעיות נסתרות (hidden issues)
    const hiddenIssues = await this.detectHiddenProblems(rawData.reviews);
    
    // 3️⃣ חיזוי עתידי (predictive analysis)
    const predictions = await this.predictFutureIssues(productName, category, rawData);
    
    // 4️⃣ השוואה חכמה (not just specs)
    const smartComparison = await this.compareIntelligently(productName, category);
    
    // 5️⃣ המלצה מותאמת אישית
    const personalizedRecommendation = await this.generatePersonalizedAdvice(
      rawData,
      patternAnalysis,
      hiddenIssues
    );
    
    return {
      // ✅ ערך: זיהוי דפוסים שמשתמש רגיל לא יראה
      patterns: {
        timeBasedIssues: patternAnalysis.timeBasedIssues,
        // "Battery issues appear after 6 months (mentioned in 234 reviews)"
        
        correlations: patternAnalysis.correlations,
        // "Users who mention 'gaming' also report 'heating' (87% correlation)"
        
        sentiment_trend: patternAnalysis.sentimentTrend,
        // "Sentiment declining over time: 4.5★ (launch) → 3.8★ (now)"
        
        regionalDifferences: patternAnalysis.regional
        // "EU units have 2x higher failure rate than US units"
      },
      
      // ✅ ערך: בעיות שלא כתובות במפורש
      hiddenIssues: {
        implicitProblems: hiddenIssues.implicit,
        // "Many reviews mention 'works fine' BUT also 'returned it' - red flag"
        
        euphemisms: hiddenIssues.euphemisms,
        // "34 reviews say 'not bad' - usually means disappointed"
        
        omissions: hiddenIssues.omissions
        // "No reviews mention longevity - product too new to assess"
      },
      
      // ✅ ערך: חיזוי בעיות עתידיות
      predictions: {
        likelyIssues: predictions.futureProblems,
        // "Based on similar products, expect charging port issues after 18 months"
        
        resaleValue: predictions.resale,
        // "Predicted resale value: $450 after 1 year (based on iPhone 14 data)"
        
        obsolescence: predictions.obsolescence
        // "Likely to be discontinued in 8 months (iPhone 16 launch)"
      },
      
      // ✅ ערך: השוואה מעבר למפרט טכני
      comparison: {
        realWorldPerformance: smartComparison.realWorld,
        // "Specs say 20hr battery, users report 12-14hr (vs Galaxy S24: 16hr real-world)"
        
        valueForMoney: smartComparison.value,
        // "At $799: overpriced by 18% vs comparable Android phones"
        
        ecosystemLockIn: smartComparison.ecosystem
        // "Switching cost: ~$400 (AirPods, Watch, apps) - factor this in"
      },
      
      // ✅ ערך: המלצה מותאמת לפרופיל
      personalizedAdvice: personalizedRecommendation
    };
  }

  async analyzeReviewPatterns(reviews) {
    // ניתוח 13,000 ביקורות - זיהוי דפוסים שמשתמש רגיל לא יראה
    
    // דוגמה: ניתוח sentiment לפי זמן
    const reviewsByMonth = this.groupByMonth(reviews);
    const sentimentTrend = reviewsByMonth.map(month => ({
      month: month.date,
      avgRating: month.avgRating,
      volume: month.count
    }));
    
    // דוגמה: זיהוי בעיות שמופיעות אחרי X זמן
    const timeBasedIssues = this.analyzeTimeBasedIssues(reviews);
    // "Battery issues spike at 6-month mark"
    
    // דוגמה: קורלציות בין תכונות
    const correlations = this.findCorrelations(reviews);
    // "Users who mention 'camera' have 23% higher satisfaction"
    
    return {
      sentimentTrend,
      timeBasedIssues,
      correlations,
      regional: this.analyzeByRegion(reviews)
    };
  }

  async detectHiddenProblems(reviews) {
    // זיהוי בעיות לא מפורשות
    
    const implicit = [];
    
    // דוגמה: "works fine" + "returned" = red flag
    const contradictoryReviews = reviews.filter(r => 
      r.text.includes('works fine') && r.text.includes('return')
    );
    
    if (contradictoryReviews.length > 50) {
      implicit.push({
        issue: "Contradiction pattern detected",
        description: `${contradictoryReviews.length} reviews say "works fine" but returned it`,
        severity: "medium"
      });
    }
    
    // דוגמה: euphemisms
    const euphemisms = this.detectEuphemisms(reviews);
    // "not bad", "okay", "acceptable" = usually disappointed
    
    // דוגמה: suspicious 5-star reviews
    const suspiciousPraise = reviews.filter(r => 
      r.rating === 5 && r.text.length < 50
    );
    
    return { implicit, euphemisms, suspiciousPraise };
  }

  async predictFutureIssues(productName, category, rawData) {
    // חיזוי על בסיס היסטוריה של מוצרים דומים
    
    // דוגמה: iPhone 15 → look at iPhone 14/13 patterns
    const historicalData = await this.getHistoricalData(category, "iPhone");
    
    const predictions = {
      futureProblems: [],
      resale: this.predictResaleValue(productName, historicalData),
      obsolescence: this.predictObsolescence(productName)
    };
    
    // "iPhone 14 had charging port issues after 18 months → expect same"
    if (historicalData.commonIssues.includes('charging_port')) {
      predictions.futureProblems.push({
        issue: "Charging port failure",
        likelihood: "high",
        expectedTimeframe: "18-24 months",
        basedOn: "iPhone 14 pattern"
      });
    }
    
    return predictions;
  }

  async compareIntelligently(productName, category) {
    // השוואה מבוססת real-world data, לא רק specs
    
    return {
      realWorld: {
        batteryLife: {
          claimed: "20 hours",
          actual: "12-14 hours (based on 1,234 user reports)",
          competitors: [
            { name: "Galaxy S24", claimed: "22hr", actual: "16hr" },
            { name: "Pixel 8", claimed: "24hr", actual: "18hr" }
          ],
          verdict: "Under-performs vs competition"
        }
      },
      
      value: {
        price: 799,
        fairValue: 650, // based on specs + performance
        overpriced: true,
        reason: "Similar specs available for $200 less (OnePlus 12)"
      },
      
      ecosystem: {
        lockIn: true,
        switchingCost: 400, // AirPods, Watch, apps
        warning: "Factor in $400 switching cost if leaving Apple ecosystem"
      }
    };
  }

  async generatePersonalizedAdvice(rawData, patterns, hiddenIssues) {
    // המלצה מותאמת לפי פרופיל משתמש
    
    const advice = {
      buyIf: [],
      avoidIf: [],
      alternatives: [],
      timing: ""
    };
    
    // Logic examples:
    
    // Heavy gamer?
    if (patterns.correlations.gaming_heating > 0.8) {
      advice.avoidIf.push("You're a heavy mobile gamer (overheating issues)");
      advice.alternatives.push({
        name: "ROG Phone 7",
        reason: "Better cooling system, designed for gaming"
      });
    }
    
    // Budget conscious?
    if (rawData.price > rawData.fairValue * 1.2) {
      advice.timing = "⏰ WAIT - Price 20% above fair value, expected drop in 4 weeks";
    }
    
    // Long-term keeper?
    if (hiddenIssues.implicit.some(i => i.issue.includes('durability'))) {
      advice.avoidIf.push("You plan to keep phone for 3+ years (durability concerns)");
    }
    
    return advice;
  }
}

export default ExpertAnalyzer;
