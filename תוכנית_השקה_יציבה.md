import React from 'react';

class ErrorBoundary extends React.Component {
  constructor(props) {
    super(props);
    this.state = { hasError: false, error: null };
  }

  static getDerivedStateFromError(error) {
    return { hasError: true, error };
  }

  componentDidCatch(error, errorInfo) {
    console.error('Error caught by boundary:', error, errorInfo);
    // TODO: Send to error tracking service (Sentry)
  }

  render() {
    if (this.state.hasError) {

          <button onClick={() => window.location.reload()}>
            רענן דף
          </button>
        </div>
      );
    }

    return this.props.children;
  }
}

export default ErrorBoundary;

#### 3. Redis/Cache Stability

**3.1 Redis Connection Handling**

```javascript
// backend/config/redis.js
const redis = require('redis');
const logger = require('../utils/logger');

let client = null;

const getRedisClient = async () => {
  if (client) return client;
  
  try {
    client = redis.createClient({
      url: process.env.REDIS_URL,
      socket: {
        reconnectStrategy: (retries) => {
          if (retries > 10) {
            logger.error('Redis: Too many reconnect attempts');
            return new Error('Redis unavailable');
          }
          return Math.min(retries * 100, 3000);
        }
      }
    });
    
    client.on('error', (err) => {
      logger.error('Redis error:', err);
    });
    
    client.on('connect', () => {
      logger.info('Redis connected');
    });
    
    client.on('reconnecting', () => {
      logger.warn('Redis reconnecting...');
    });
    
    await client.connect();
    return client;
    
  } catch (error) {
    logger.error('Redis connection failed:', error);
    // Fallback: return null and handle gracefully
    return null;
  }
};

// Graceful operations (fallback if Redis is down)
const get = async (key) => {
  try {
    const client = await getRedisClient();
    if (!client) return null;
    
    const value = await client.get(key);
    return value ? JSON.parse(value) : null;
  } catch (error) {
    logger.error('Redis GET failed:', error);
    return null; // Fallback to cache miss
  }
};

const set = async (key, value, ttl) => {
  try {
    const client = await getRedisClient();
    if (!client) return false;
    
    await client.set(key, JSON.stringify(value), {
      EX: ttl
    });
    return true;
  } catch (error) {
    logger.error('Redis SET failed:', error);
    return false; // System still works, just no caching
  }
};

module.exports = { get, set, getRedisClient };
```

**צריך לוודא:**
- [ ] Redis connection עם reconnect strategy
- [ ] Graceful fallback אם Redis down
- [ ] Error logging
- [ ] המערכת עובדת גם ללא Redis (slower)

**3.2 Cache Fallback בDossier Builder**

```javascript
// backend/services/simpleDossierBuilder.js

async buildDossier(productName, useCache = true) {
  let productId = null;
  
  try {
    // Try cache (but don't fail if cache is down)
    if (useCache && productId) {
      try {
        const cached = await smartCache.getDossier(productId);
        if (cached && !smartCache.isStale(cached)) {
          return { ...cached, fromCache: true };
        }
      } catch (cacheError) {
        logger.warn('Cache error (continuing without cache):', cacheError);
        // Continue without cache - not critical
      }
    }
    
    // Build dossier (main logic)
    const result = await this.buildDossierCore(productName);
    
    // Try to cache result (but don't fail if cache fails)
    if (useCache) {
      try {
        await smartCache.saveDossier(productId, result);
      } catch (cacheError) {
        logger.warn('Failed to cache result (not critical):', cacheError);
        // Don't fail - result is still valid
      }
    }
    
    return result;
    
  } catch (error) {
    logger.error('Build failed:', error);
    throw error; // This is a real error
  }
}
```

**צריך לוודא:**
- [ ] Cache errors לא קורסים את המערכת
- [ ] Logger מתריע על cache issues
- [ ] המערכת עובדת ללא cache (אך יותר לאט)
